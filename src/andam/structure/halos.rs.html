1//! Halo mass function and bias

3use crate::constants::*;
4use std::f64::consts::PI;
5
6/// Halo mass function formalism
7#[derive(Debug, Clone, Copy)]
8pub enum MassFunctionType {
9    PressSchechter,
10    ShethTormen,
11    Tinker,
12}
13
14/// Halo mass function calculator
15pub struct HaloMassFunction {
16    pub omega_m: f64,
17    pub sigma_8: f64,
18    pub mf_type: MassFunctionType,
19}
20
21impl HaloMassFunction {
22    /// Create new mass function calculator
23    pub fn new(omega_m: f64, sigma_8: f64, mf_type: MassFunctionType) -> Self {
24        HaloMassFunction {
25            omega_m,
26            sigma_8,
27            mf_type,
28        }
29    }
30
31    /// RMS fluctuation σ(M) at mass scale M
32    pub fn sigma_mass(&self, mass_msun: f64, z: f64) -> f64 {
33        // σ(M) ∝ M^(-1/3) approximately
34        // Normalized so σ(M*) = σ_8 at M* ≈ 10^13 M_sun
35        let m_star = 1e13;
36        let sigma_star = self.sigma_8;
37
38        // Power law scaling
39        let sigma = sigma_star * (mass_msun / m_star).powf(-1.0 / 3.0);
40
41        // Redshift evolution: σ(z) = σ(0) / D(z)
42        // Simplified: D(z) ≈ 1/(1+z) in matter era
43        sigma / (1.0 + z)
44    }
45
46    /// Peak height ν = δ_c / σ(M)
47    pub fn peak_height(&self, mass_msun: f64, z: f64) -> f64 {
48        let delta_c = 1.686; // Critical overdensity for collapse
49        let sigma = self.sigma_mass(mass_msun, z);
50        delta_c / sigma
51    }
52
53    /// Multiplicity function f(ν)
54    pub fn multiplicity_function(&self, nu: f64) -> f64 {
55        match self.mf_type {
56            MassFunctionType::PressSchechter => {
57                // f(ν) = √(2/π) ν exp(-ν²/2)
58                (2.0 / PI).sqrt() * nu * (-nu.powi(2) / 2.0).exp()
59            }
60            MassFunctionType::ShethTormen => {
61                // Sheth-Tormen (1999)
62                let a = 0.707;
63                let p = 0.3;
64                let a_nu2 = a * nu.powi(2);
65
66                let a_factor = (2.0 * a / PI).sqrt();
67                a_factor * (1.0 + a_nu2.powf(-p)) * (a_nu2).sqrt() * (-a_nu2 / 2.0).exp()
68            }
69            MassFunctionType::Tinker => {
70                // Tinker et al. (2008) - simplified
71                let alpha = 0.368;
72                let beta = 0.589;
73                let gamma = 0.864;
74                let phi = -0.729;
75
76                alpha
77                    * (1.0 + (beta / nu).powf(2.0 * phi))
78                    * nu.powf(2.0 * beta)
79                    * (-gamma * nu.powi(2) / 2.0).exp()
80            }
81        }
82    }
83
84    /// Halo mass function dn/dM [h⁴ Mpc⁻³ M_sun⁻¹]
85    pub fn dn_dm(&self, mass_msun: f64, z: f64) -> f64 {
86        // dn/dM = (ρ̄/M) f(ν) |dlnσ/dlnM|
87        let rho_crit = critical_density(70.0); // kg/m³
88        let rho_m = self.omega_m * rho_crit;
89        let rho_m_msun_mpc3 = rho_m * (PARSEC * 1e6).powi(3) / M_SUN;
90
91        let nu = self.peak_height(mass_msun, z);
92        let f_nu = self.multiplicity_function(nu);
93
94        let sigma = self.sigma_mass(mass_msun, z);
95        let dln_sigma_dln_m = 1.0 / 3.0; // For σ ∝ M^(-1/3)
96
97        (rho_m_msun_mpc3 / mass_msun) * f_nu * (dln_sigma_dln_m / sigma)
98    }
99
100    /// Number density of halos in mass range [M_min, M_max]
101    pub fn number_density(&self, m_min: f64, m_max: f64, z: f64, n_bins: usize) -> f64 {
102        let log_m_min = m_min.ln();
103        let log_m_max = m_max.ln();
104        let dlogm = (log_m_max - log_m_min) / n_bins as f64;
105
106        let mut n_total = 0.0;
107
108        for i in 0..n_bins {
109            let log_m = log_m_min + (i as f64 + 0.5) * dlogm;
110            let m = log_m.exp();
111            let dn_dm = self.dn_dm(m, z);
112            n_total += dn_dm * m * dlogm;
113        }
114
115        n_total
116    }
117
118    /// Halo bias b(M)
119    pub fn halo_bias(&self, mass_msun: f64, z: f64) -> f64 {
120        let nu = self.peak_height(mass_msun, z);
121        let delta_c = 1.686;
122
123        match self.mf_type {
124            MassFunctionType::PressSchechter => {
125                // b = 1 + (ν² - 1)/δ_c
126                1.0 + (nu.powi(2) - 1.0) / delta_c
127            }
128            MassFunctionType::ShethTormen => {
129                // Sheth-Tormen bias
130                let a = 0.707;
131                let b_param = 0.5;
132                let c = 0.6;
133
134                let a_nu2 = a * nu.powi(2);
135                1.0 + (a_nu2 - 1.0) / delta_c + 2.0 * b_param / (delta_c * (1.0 + a_nu2.powf(c)))
136            }
137            MassFunctionType::Tinker => {
138                // Tinker et al. (2010) simplified bias
139                let y = (nu.powi(2)).max(0.01);
140                1.0 + 0.24 * y.ln() + 0.4 * nu.powi(2)
141            }
142        }
143    }
144}

146#[cfg(test)]
147mod tests {
148    use super::*;
149
150    #[test]
151    fn test_halo_mass_function() {
152        let hmf = HaloMassFunction::new(0.3, 0.8, MassFunctionType::ShethTormen);
153
154        let m = 1e14; // M_sun
155        let z = 0.0;
156
157        let dn_dm = hmf.dn_dm(m, z);
158        assert!(dn_dm > 0.0);
159    }
160
161    #[test]
162    fn test_halo_bias_increases_with_mass() {
163        let hmf = HaloMassFunction::new(0.3, 0.8, MassFunctionType::ShethTormen);
164
165        let b1 = hmf.halo_bias(1e12, 0.0);
166        let b2 = hmf.halo_bias(1e15, 0.0);
167
168        // More massive halos should be more biased
169        assert!(b2 > b1, "b1 = {}, b2 = {}", b1, b2);
170    }
171
172    #[test]
173    fn test_sigma_mass_decreases_with_mass() {
174        let hmf = HaloMassFunction::new(0.3, 0.8, MassFunctionType::ShethTormen);
175
176        let s1 = hmf.sigma_mass(1e12, 0.0);
177        let s2 = hmf.sigma_mass(1e14, 0.0);
178
179        assert!(s2 < s1);
180    }
