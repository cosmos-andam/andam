6/// Primordial power spectrum from inflation
7///
8/// P(k) = A_s (k/k_pivot)^(n_s - 1)
9///
10/// # Arguments
11/// * `k` - Wavenumber [h/Mpc]
12/// * `amplitude` - A_s (typically ~2.1e-9)
13/// * `spectral_index` - n_s (typically ~0.96)
14/// * `k_pivot` - Pivot scale [h/Mpc] (typically 0.05)
15pub fn primordial_power_spectrum(k: f64, amplitude: f64, spectral_index: f64, k_pivot: f64) -> f64 {
16    amplitude * (k / k_pivot).powf(spectral_index - 1.0)
17}

19/// Transfer function (simplified Eisenstein & Hu form)
20///
21/// T(k) suppresses power on small scales
22pub fn transfer_function_eh(k: f64, omega_m: f64, omega_b: f64, h: f64) -> f64 {
23    // Eisenstein & Hu (1998) fitting formula
24    let omega_m_h2 = omega_m * h * h;
25    let omega_b_h2 = omega_b * h * h;
26    let theta_cmb = T_CMB / 2.7;
27
28    // Sound horizon (not used in simplified version)
29    let _s =
30        44.5 * (omega_m_h2 / theta_cmb.powi(4)).ln() / (1.0 + 10.0 * omega_b_h2.powf(0.75)).sqrt();
31
32    // Silk damping scale (not used in simplified version)
33    let _k_silk = 1.6
34        * omega_b_h2.powf(0.52)
35        * omega_m_h2.powf(0.73)
36        * (1.0 + (10.4 * omega_m_h2).powf(-0.95));
37
38    let q = k / (13.41 * omega_m_h2.powf(0.5));
39    let l = (2.0 * std::f64::consts::E + 1.8 * q).ln();
40    let c = 14.2 + 731.0 / (1.0 + 62.5 * q);
41
42    l / (l + c * q * q)
43}

45/// Linear matter power spectrum P(k)
46///
47/// P(k) = A_s (k/k_pivot)^(n_s-1) T²(k) D²(z) (2π²/k³)
48pub fn matter_power_spectrum(
49    k: f64,
50    z: f64,
51    omega_m: f64,
52    omega_b: f64,
53    h: f64,
54    amplitude: f64,
55    spectral_index: f64,
56) -> f64 {
57    let p_prim = primordial_power_spectrum(k, amplitude, spectral_index, 0.05);
58    let t = transfer_function_eh(k, omega_m, omega_b, h);
59
60    // Growth factor (simplified for matter-dominated)
61    let d = 1.0 / (1.0 + z);
62
63    // Dimensional factor
64    let factor = 2.0 * PI * PI / (k * k * k);
65
66    p_prim * t * t * d * d * factor
67}

69/// Dimensionless power spectrum Δ²(k)
70pub fn dimensionless_power(
71    k: f64,
72    z: f64,
73    omega_m: f64,
74    omega_b: f64,
75    h: f64,
76    amplitude: f64,
77    spectral_index: f64,
78) -> f64 {
79    let p_k = matter_power_spectrum(k, z, omega_m, omega_b, h, amplitude, spectral_index);
80    k * k * k * p_k / (2.0 * PI * PI)
81}

83#[cfg(test)]
84mod tests {
85    use super::*;

87    #[test]
88    fn test_primordial_power_at_pivot() {
89        let p = primordial_power_spectrum(0.05, 2.1e-9, 0.96, 0.05);
90        assert!((p - 2.1e-9).abs() < 1e-15);
91    }
92
93    #[test]
94    fn test_transfer_function_limiting_cases() {
95        // Should be ~1 at large scales (small k)
96        let t_large = transfer_function_eh(1e-4, 0.3, 0.05, 0.7);
97        assert!(t_large > 0.9 && t_large <= 1.0);
98
99        // Should be <1 at small scales (large k)
100        let t_small = transfer_function_eh(10.0, 0.3, 0.05, 0.7);
101        assert!(t_small < 0.5);
102    }
