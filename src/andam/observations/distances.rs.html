1//! Cosmological distance measures
2//!
3//! Implements various distance definitions used in cosmology:
4//! - Comoving distance
5//! - Luminosity distance
6//! - Angular diameter distance
7//! - Distance modulus

12/// Integrand for comoving distance: 1/H(z)
13fn comoving_integrand(z: f64, universe: &Universe) -> f64 {
14    1.0 / universe.hubble_z(z)
15}
16
17/// Comoving distance to redshift z \[Mpc\]
18///
19/// d_c = (c/H_0) ∫₀^z dz'/H(z')
20pub fn comoving_distance(z: f64, universe: &Universe) -> f64 {
21    let c_over_h0 = C / (universe.h0 * 1e3) * PARSEC * 1e-6; // Mpc
22
23    // Simpson's rule integration
24    let n = 1000;
25    let dz = z / n as f64;
26    let mut sum = comoving_integrand(0.0, universe) + comoving_integrand(z, universe);
27
28    for i in 1..n {
29        let zi = i as f64 * dz;
30        let weight = if i % 2 == 0 { 2.0 } else { 4.0 };
31        sum += weight * comoving_integrand(zi, universe);
32    }
33
34    c_over_h0 * sum * dz / 3.0
35}

37/// Transverse comoving distance \[Mpc\]
38///
39/// Accounts for curvature:
40/// - Flat: d_M = d_c
41/// - Open: d_M = (c/H_0)/√|Ω_k| sinh(√|Ω_k| d_c H_0/c)
42/// - Closed: d_M = (c/H_0)/√Ω_k sin(√Ω_k d_c H_0/c)
43pub fn transverse_comoving_distance(z: f64, universe: &Universe) -> f64 {
44    let d_c = comoving_distance(z, universe);
45    let omega_k = 1.0 - universe.omega_total();
46    let c_over_h0 = C / (universe.h0 * 1e3) * PARSEC * 1e-6;
47
48    if omega_k.abs() < 1e-5 {
49        // Flat universe
50        d_c
51    } else if omega_k > 0.0 {
52        // Open universe
53        let sqrt_ok = omega_k.sqrt();
54        (c_over_h0 / sqrt_ok) * (sqrt_ok * d_c / c_over_h0).sinh()
55    } else {
56        // Closed universe
57        let sqrt_ok = (-omega_k).sqrt();
58        (c_over_h0 / sqrt_ok) * (sqrt_ok * d_c / c_over_h0).sin()
59    }
60}

62/// Luminosity distance \[Mpc\]
63///
64/// d_L = (1+z) d_M
65pub fn luminosity_distance(z: f64, universe: &Universe) -> f64 {
66    (1.0 + z) * transverse_comoving_distance(z, universe)
67}
68
69/// Angular diameter distance \[Mpc\]
70///
71/// d_A = d_M / (1+z)
72pub fn angular_diameter_distance(z: f64, universe: &Universe) -> f64 {
73    transverse_comoving_distance(z, universe) / (1.0 + z)
74}
75
76/// Distance modulus (m - M)
77///
78/// μ = 5 log₁₀(d_L / 10 pc)
79pub fn distance_modulus(z: f64, universe: &Universe) -> f64 {
80    let d_l = luminosity_distance(z, universe);
81    let d_l_pc = d_l * 1e6; // Convert Mpc to pc
82    5.0 * (d_l_pc / 10.0).log10()
83}
84
85/// Comoving volume element dV_c/dz/dΩ [Mpc³/sr]
86pub fn comoving_volume_element(z: f64, universe: &Universe) -> f64 {
87    let d_h = C / (universe.h0 * 1e3) * PARSEC * 1e-6; // Hubble distance \[Mpc\]
88    let d_m = transverse_comoving_distance(z, universe);
89    let e_z = universe.hubble_z(z) / universe.h0;
90
91    d_h * d_m * d_m / e_z
92}

94#[cfg(test)]
95mod tests {
96    use super::*;
97    use approx::assert_relative_eq;
98
99    #[test]
100    fn test_comoving_distance_z0() {
101        let universe = Universe::benchmark();
102        let d = comoving_distance(0.0, &universe);
103        assert_relative_eq!(d, 0.0, epsilon = 1e-6);
104    }
105
106    #[test]
107    fn test_distance_relations() {
108        let universe = Universe::benchmark();
109        let z = 1.0;
110
111        let d_l = luminosity_distance(z, &universe);
112        let d_a = angular_diameter_distance(z, &universe);
113
114        // d_L = (1+z)² d_A
115        assert_relative_eq!(d_l, (1.0 + z).powi(2) * d_a, epsilon = 1e-6);
116    }
117
118    #[test]
119    fn test_flat_universe_distances() {
120        let universe = Universe::benchmark();
121        let z = 0.5;
122
123        let d_c = comoving_distance(z, &universe);
124        let d_m = transverse_comoving_distance(z, &universe);
125
126        // Should be equal for flat universe (within numerical precision)
127        assert_relative_eq!(d_c, d_m, max_relative = 1e-6);
128    }
