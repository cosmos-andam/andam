1//! Massive neutrino cosmology
2
3/// Neutrino mass hierarchy
4#[derive(Debug, Clone, Copy)]
5pub enum MassHierarchy {
6    Normal,
7    Inverted,
8    Degenerate,
9}
10
11/// Massive neutrino component
12pub struct MassiveNeutrinos {
13    pub total_mass: f64, // Sum of masses [eV]
14    pub hierarchy: MassHierarchy,
15    pub n_species: usize,
16}
17
18impl MassiveNeutrinos {
19    /// Create neutrino model
20    pub fn new(total_mass: f64, hierarchy: MassHierarchy) -> Self {
21        MassiveNeutrinos {
22            total_mass,
23            hierarchy,
24            n_species: 3,
25        }
26    }
27
28    /// Neutrino density parameter
29    pub fn omega_nu(&self, h: f64) -> f64 {
30        // Ω_ν h² = Σm_ν / (93.14 eV)
31        self.total_mass / 93.14 / (h * h)
32    }
33
34    /// Suppression of matter power spectrum
35    pub fn power_suppression(&self, k: f64, _z: f64) -> f64 {
36        // Simplified: exp(-(k/k_fs)^α) where k_fs depends on neutrino mass
37        // For m_ν ~ 0.06 eV, k_fs ~ 0.05-0.1 h/Mpc
38        let k_fs = 0.3 * (self.total_mass / 0.06); // Free-streaming scale [h/Mpc]
39        let alpha = 1.5; // Power-law suppression
40        (-(k / k_fs).powf(alpha)).exp()
41    }
42}

44#[cfg(test)]
45mod tests {
46    use super::*;
47
48    #[test]
49    fn test_neutrino_omega() {
50        let neutrinos = MassiveNeutrinos::new(0.06, MassHierarchy::Normal);
51        let omega_nu = neutrinos.omega_nu(0.7);
52
53        // Should be small
54        assert!(omega_nu < 0.01);
55        assert!(omega_nu > 0.0);
56    }
57
58    #[test]
59    fn test_power_suppression() {
60        let neutrinos = MassiveNeutrinos::new(0.06, MassHierarchy::Normal);
61
62        // At small k, minimal suppression
63        let s_small = neutrinos.power_suppression(0.01, 0.0);
64        assert!(s_small > 0.9);
65
66        // At large k, significant suppression
67        let s_large = neutrinos.power_suppression(1.0, 0.0);
68        assert!(s_large < 0.5);
69    }
