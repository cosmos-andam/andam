1//! Friedmann equation solver and universe evolution
2
3use crate::constants::*;
4use crate::dynamics::components::Component;
5
6/// Represents a cosmological model with multiple components
7#[derive(Debug, Clone)]
8pub struct Universe {
9    /// Hubble constant today [km/s/Mpc]
10    pub h0: f64,
11    /// Components (matter, radiation, dark energy, curvature)
12    pub components: Vec<Component>,
13}
14
15impl Universe {
16    /// Create a new universe with given Hubble constant
17    pub fn new(h0: f64) -> Self {
18        Universe {
19            h0,
20            components: Vec::new(),
21        }
22    }
23
24    /// Add a component to the universe
25    pub fn add_component(&mut self, component: Component) {
26        self.components.push(component);
27    }
28
29    /// Create a benchmark ΛCDM model (Planck 2018)
30    pub fn benchmark() -> Self {
31        let mut universe = Universe::new(67.66);
32        universe.add_component(Component::matter(0.3111));
33        universe.add_component(Component::radiation(9.3e-5));
34        universe.add_component(Component::dark_energy(0.6889));
35        universe
36    }
37
38    /// Create an Einstein-de Sitter universe (Ω_m = 1, flat)
39    pub fn einstein_de_sitter(h0: f64) -> Self {
40        let mut universe = Universe::new(h0);
41        universe.add_component(Component::matter(1.0));
42        universe
43    }
44
45    /// Total Omega (should be 1 for flat universe)
46    pub fn omega_total(&self) -> f64 {
47        self.components.iter().map(|c| c.omega_0).sum()
48    }
49
50    /// Hubble parameter at scale factor a: H(a) / H_0
51    pub fn hubble_normalized(&self, a: f64) -> f64 {
52        let sum: f64 = self.components.iter().map(|c| c.density_evolution(a)).sum();
53        sum.sqrt()
54    }
55
56    /// Hubble parameter at scale factor a [km/s/Mpc]
57    pub fn hubble(&self, a: f64) -> f64 {
58        self.h0 * self.hubble_normalized(a)
59    }
60
61    /// Hubble parameter at redshift z [km/s/Mpc]
62    pub fn hubble_z(&self, z: f64) -> f64 {
63        let a = 1.0 / (1.0 + z);
64        self.hubble(a)
65    }
66
67    /// Deceleration parameter q(a) = -ä·a/ȧ²
68    pub fn deceleration(&self, a: f64) -> f64 {
69        let mut sum_rho_1_w = 0.0;
70        let mut sum_rho = 0.0;
71
72        for component in &self.components {
73            let rho = component.density_evolution(a);
74            sum_rho += rho;
75            sum_rho_1_w += rho * (1.0 + component.w);
76        }
77
78        0.5 * sum_rho_1_w / sum_rho
79    }
80
81    /// Age of universe at scale factor a \[Gyr\]
82    pub fn age(&self, a: f64) -> f64 {
83        // Integrate t = ∫ da / (a·H(a))
84        let n_steps = 1000;
85        let da = a / n_steps as f64;
86
87        let mut age = 0.0;
88        for i in 1..=n_steps {
89            let a_i = i as f64 * da;
90            let h = self.hubble(a_i); // km/s/Mpc
91            let h_si = h * 1e3 / (PARSEC * 1e6); // Convert to SI
92            let dt = da / (a_i * h_si); // dt = da/(a*H) in seconds
93            age += dt;
94        }
95
96        age / GYR // Convert to Gyr
97    }
98
99    /// Age of universe today \[Gyr\]
100    pub fn age_today(&self) -> f64 {
101        self.age(1.0)
102    }
103
104    /// Critical density today [kg/m³]
105    pub fn critical_density(&self) -> f64 {
106        critical_density(self.h0)
107    }
108}

110#[cfg(test)]
111mod tests {
112    use super::*;
113    use approx::assert_relative_eq;
114
115    #[test]
116    fn test_benchmark_universe() {
117        let universe = Universe::benchmark();
118        let omega_tot = universe.omega_total();
119        assert_relative_eq!(omega_tot, 1.0, epsilon = 1e-3);
120    }
121
122    #[test]
123    fn test_einstein_de_sitter() {
124        let universe = Universe::einstein_de_sitter(70.0);
125        assert_relative_eq!(universe.omega_total(), 1.0, epsilon = 1e-10);
126
127        // H(a) = H_0 * a^(-3/2) for matter-only
128        assert_relative_eq!(universe.hubble_normalized(0.5), 2.828427, epsilon = 1e-5);
129    }
130
131    #[test]
132    fn test_age_calculation() {
133        let universe = Universe::benchmark();
134        let age = universe.age_today();
135        // Should be around 13.8 Gyr
136        assert!(age > 13.0 && age < 14.5);
137    }
