3/// Stokes parameters for polarization
4#[derive(Debug, Clone, Copy)]
5pub struct StokesParameters {
6    pub q: f64, // Linear polarization (Q)
7    pub u: f64, // Linear polarization (U)
8}
9
10impl StokesParameters {
11    /// Polarization fraction P = √(Q² + U²)/I
12    pub fn polarization_fraction(&self, intensity: f64) -> f64 {
13        ((self.q.powi(2) + self.u.powi(2)).sqrt()) / intensity
14    }
15
16    /// Polarization angle χ = 0.5 arctan(U/Q)
17    pub fn angle(&self) -> f64 {
18        0.5 * self.u.atan2(self.q)
19    }
20}

24    // In full implementation, use spin-weighted spherical harmonics
25    // This is simplified
26
27    let mut e_map = vec![0.0; q_map.len()];
28    let mut b_map = vec![0.0; u_map.len()];
29
30    e_map.copy_from_slice(q_map);
31    b_map.copy_from_slice(u_map);

36/// E-mode and B-mode power spectra
37pub struct PolarizationSpectrum {
38    pub l_max: usize,
39    pub c_l_ee: Vec<f64>, // E-mode auto
40    pub c_l_bb: Vec<f64>, // B-mode auto
41    pub c_l_te: Vec<f64>, // Temperature-E cross
42}
43
44impl PolarizationSpectrum {
45    /// Create new polarization spectrum
46    pub fn new(l_max: usize) -> Self {
47        PolarizationSpectrum {
48            l_max,
49            c_l_ee: vec![0.0; l_max + 1],
50            c_l_bb: vec![0.0; l_max + 1],
51            c_l_te: vec![0.0; l_max + 1],
52        }
53    }
54
55    /// Compute from Boltzmann evolution
56    pub fn from_boltzmann(_universe: &crate::dynamics::Universe, l_max: usize) -> Self {
57        let mut spectrum = Self::new(l_max);
58
59        // Compute C_l^EE, C_l^BB, C_l^TE
60        // Full implementation requires Boltzmann solver
61
62        for l in 2..=l_max {
63            // E-mode power (scalar perturbations)
64            spectrum.c_l_ee[l] = 5000.0 * (l as f64 / 1000.0).powi(-1);
65
66            // B-mode power (tensor perturbations only)
67            // Tiny for standard inflation, zero for scalars
68            spectrum.c_l_bb[l] = 0.001 * (l as f64 / 100.0).powi(-2);
69
70            // TE cross-correlation
71            spectrum.c_l_te[l] = -50.0 * (l as f64 / 100.0).powi(-1);
72        }
73
74        spectrum
75    }
76
77    /// Tensor-to-scalar ratio constraint from B-modes
78    pub fn tensor_to_scalar_ratio(&self) -> f64 {
79        // r = C_l^BB(tensor) / C_l^TT(scalar)
80        // Simplified calculation
81        let l_pivot = 80;
82        if l_pivot < self.c_l_bb.len() {
83            self.c_l_bb[l_pivot] / 5000.0
84        } else {
85            0.0
86        }
87    }
88}
