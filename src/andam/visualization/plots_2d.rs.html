6/// Configuration for 2D plots
7#[derive(Debug, Clone)]
8pub struct PlotConfig {
9    pub title: String,
10    pub x_label: String,
11    pub y_label: String,
12    pub width: u32,
13    pub height: u32,
14    pub x_log: bool,
15    pub y_log: bool,
16}
17
18impl Default for PlotConfig {
19    fn default() -> Self {
20        PlotConfig {
21            title: "Plot".to_string(),
22            x_label: "x".to_string(),
23            y_label: "y".to_string(),
24            width: 1024,
25            height: 768,
26            x_log: false,
27            y_log: false,
28        }
29    }
30}

32/// Create a simple line plot
33pub fn create_line_plot(
34    filename: &str,
35    data: &[(f64, f64)],
36    config: &PlotConfig,
37) -> Result<(), Box<dyn Error>> {
38    let root = BitMapBackend::new(filename, (config.width, config.height)).into_drawing_area();
39    root.fill(&WHITE)?;
40
41    // Find data ranges
42    let x_min = data.iter().map(|(x, _)| *x).fold(f64::INFINITY, f64::min);
43    let x_max = data
44        .iter()
45        .map(|(x, _)| *x)
46        .fold(f64::NEG_INFINITY, f64::max);
47    let y_min = data.iter().map(|(_, y)| *y).fold(f64::INFINITY, f64::min);
48    let y_max = data
49        .iter()
50        .map(|(_, y)| *y)
51        .fold(f64::NEG_INFINITY, f64::max);
52
53    let mut chart = ChartBuilder::on(&root)
54        .caption(&config.title, ("sans-serif", 40).into_font())
55        .margin(10)
56        .x_label_area_size(40)
57        .y_label_area_size(50)
58        .build_cartesian_2d(x_min..x_max, y_min..y_max)?;
59
60    chart
61        .configure_mesh()
62        .x_desc(&config.x_label)
63        .y_desc(&config.y_label)
64        .draw()?;
65
66    chart.draw_series(LineSeries::new(data.iter().map(|(x, y)| (*x, *y)), &BLUE))?;
67
68    root.present()?;
69    Ok(())
70}

72/// Create a log-log plot
73pub fn create_loglog_plot(
74    filename: &str,
75    data: &[(f64, f64)],
76    config: &PlotConfig,
77) -> Result<(), Box<dyn Error>> {
78    let root = BitMapBackend::new(filename, (config.width, config.height)).into_drawing_area();
79    root.fill(&WHITE)?;
80
81    // Find data ranges (positive values only for log)
82    let x_min = data
83        .iter()
84        .map(|(x, _)| *x)
85        .filter(|x| *x > 0.0)
86        .fold(f64::INFINITY, f64::min);
87    let x_max = data
88        .iter()
89        .map(|(x, _)| *x)
90        .filter(|x| *x > 0.0)
91        .fold(f64::NEG_INFINITY, f64::max);
92    let y_min = data
93        .iter()
94        .map(|(_, y)| *y)
95        .filter(|y| *y > 0.0)
96        .fold(f64::INFINITY, f64::min);
97    let y_max = data
98        .iter()
99        .map(|(_, y)| *y)
100        .filter(|y| *y > 0.0)
101        .fold(f64::NEG_INFINITY, f64::max);
102
103    let mut chart = ChartBuilder::on(&root)
104        .caption(&config.title, ("sans-serif", 40).into_font())
105        .margin(10)
106        .x_label_area_size(40)
107        .y_label_area_size(50)
108        .build_cartesian_2d((x_min..x_max).log_scale(), (y_min..y_max).log_scale())?;
109
110    chart
111        .configure_mesh()
112        .x_desc(&config.x_label)
113        .y_desc(&config.y_label)
114        .draw()?;
115
116    chart.draw_series(LineSeries::new(
117        data.iter()
118            .filter(|(x, y)| *x > 0.0 && *y > 0.0)
119            .map(|(x, y)| (*x, *y)),
120        &RED,
121    ))?;
122
123    root.present()?;
124    Ok(())
125}

127/// Create a multi-line plot with legend
128pub fn create_multiline_plot(
129    filename: &str,
130    datasets: &[(&[(f64, f64)], &str)],
131    config: &PlotConfig,
132) -> Result<(), Box<dyn Error>> {
133    let root = BitMapBackend::new(filename, (config.width, config.height)).into_drawing_area();
134    root.fill(&WHITE)?;
135
136    // Find global data ranges
137    let mut x_min = f64::INFINITY;
138    let mut x_max = f64::NEG_INFINITY;
139    let mut y_min = f64::INFINITY;
140    let mut y_max = f64::NEG_INFINITY;
141
142    for (data, _) in datasets {
143        for (x, y) in *data {
144            x_min = x_min.min(*x);
145            x_max = x_max.max(*x);
146            y_min = y_min.min(*y);
147            y_max = y_max.max(*y);
148        }
149    }
150
151    let mut chart = ChartBuilder::on(&root)
152        .caption(&config.title, ("sans-serif", 40).into_font())
153        .margin(10)
154        .x_label_area_size(40)
155        .y_label_area_size(50)
156        .build_cartesian_2d(x_min..x_max, y_min..y_max)?;
157
158    chart
159        .configure_mesh()
160        .x_desc(&config.x_label)
161        .y_desc(&config.y_label)
162        .draw()?;
163
164    let colors = [&BLUE, &RED, &GREEN, &MAGENTA, &CYAN];
165
166    for (i, (data, label)) in datasets.iter().enumerate() {
167        let color = colors[i % colors.len()];
168        chart
169            .draw_series(LineSeries::new(data.iter().map(|(x, y)| (*x, *y)), color))?
170            .label(*label)
171            .legend(move |(x, y)| PathElement::new(vec![(x, y), (x + 20, y)], color));
172    }
173
174    chart.configure_series_labels().border_style(BLACK).draw()?;
175
176    root.present()?;
177    Ok(())
178}
