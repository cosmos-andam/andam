1//! Neutron-proton freeze-out calculations
2
3use crate::constants::*;
4
5/// Weak interaction rate [s^-1] at temperature T \[K\]
6pub fn weak_interaction_rate(temp_k: f64) -> f64 {
7    // Simplified formula for n <-> p + e + ν_e weak interaction rate
8    let t_mev = K_B * temp_k * J_TO_EV * 1e-6; // Temperature in MeV
9
10    // Rate proportional to T^5
11    1.13 * t_mev.powi(5)
12}
13
14/// Hubble parameter [s^-1] at temperature T \[K\] in radiation-dominated era
15pub fn hubble_rate_radiation(temp_k: f64) -> f64 {
16    // H = sqrt(8πGρ/3) where ρ is radiation density
17    // ρ = (π^2/30) g_* k_B^4 T^4 / (ℏ^3 c^5)
18    let g_star = 10.75; // Effective degrees of freedom (photons + e± + 3 neutrinos)
19
20    let rho = (std::f64::consts::PI.powi(2) / 30.0) * g_star * K_B.powi(4) * temp_k.powi(4)
21        / (HBAR.powi(3) * C.powi(5));
22
23    ((8.0 * std::f64::consts::PI * G * rho) / 3.0).sqrt()
24}
25
26/// Freeze-out temperature \[K\] where weak rate equals Hubble rate
27pub fn freezeout_temperature() -> f64 {
28    // Solve Γ_weak(T) = H(T)
29    // This occurs at T ~ 0.7 MeV
30    let t_freeze_mev = 0.7; // MeV
31    t_freeze_mev * 1e6 / (K_B * J_TO_EV)
32}
33
34/// Neutron-to-proton ratio at freeze-out
35pub fn freezeout_np_ratio() -> f64 {
36    let t_freeze = freezeout_temperature();
37
38    // n/p = exp(-Δm/kT)
39    let delta_m = (M_N - M_P) * C * C; // Mass difference [J]
40    let delta_m_mev = delta_m * J_TO_EV * 1e-6; // \[MeV\]
41    let t_mev = K_B * t_freeze * J_TO_EV * 1e-6;
42
43    (-delta_m_mev / t_mev).exp()
44}
45
46/// Estimate final helium-4 mass fraction from freeze-out
47pub fn estimate_yp_from_freezeout() -> f64 {
48    let np_freeze = freezeout_np_ratio();
49
50    // Almost all neutrons end up in He-4
51    // Y_p ≈ 2(n/p) / (1 + n/p)
52    2.0 * np_freeze / (1.0 + np_freeze)
53}
