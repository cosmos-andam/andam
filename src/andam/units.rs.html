1//! Unit conversion utilities for cosmology
2//!
3//! Provides conversions between common astronomical and cosmological units

7/// Length units
8#[derive(Debug, Clone, Copy, PartialEq)]
9pub enum Length {
10    Meter(f64),
11    Kilometer(f64),
12    AU(f64),
13    Parsec(f64),
14    Kiloparsec(f64),
15    Megaparsec(f64),
16    Gigaparsec(f64),
17}
18
19impl Length {
20    /// Convert to meters
21    pub fn to_meters(&self) -> f64 {
22        match self {
23            Length::Meter(x) => *x,
24            Length::Kilometer(x) => x * 1e3,
25            Length::AU(x) => x * AU,
26            Length::Parsec(x) => x * PARSEC,
27            Length::Kiloparsec(x) => x * PARSEC * 1e3,
28            Length::Megaparsec(x) => x * PARSEC * 1e6,
29            Length::Gigaparsec(x) => x * PARSEC * 1e9,
30        }
31    }
32
33    /// Convert to megaparsecs
34    pub fn to_mpc(&self) -> f64 {
35        self.to_meters() / (PARSEC * 1e6)
36    }
37
38    /// Convert to parsecs
39    pub fn to_pc(&self) -> f64 {
40        self.to_meters() / PARSEC
41    }
42}

44/// Mass units
45#[derive(Debug, Clone, Copy, PartialEq)]
46pub enum Mass {
47    Kilogram(f64),
48    SolarMass(f64),
49    EarthMass(f64),
50}
51
52impl Mass {
53    /// Convert to kilograms
54    pub fn to_kg(&self) -> f64 {
55        match self {
56            Mass::Kilogram(x) => *x,
57            Mass::SolarMass(x) => x * M_SUN,
58            Mass::EarthMass(x) => x * 5.972e24,
59        }
60    }
61
62    /// Convert to solar masses
63    pub fn to_solar_masses(&self) -> f64 {
64        self.to_kg() / M_SUN
65    }
66}

68/// Time units
69#[derive(Debug, Clone, Copy, PartialEq)]
70pub enum Time {
71    Second(f64),
72    Year(f64),
73    Kiloyear(f64),
74    Megayear(f64),
75    Gigayear(f64),
76}
77
78impl Time {
79    /// Convert to seconds
80    pub fn to_seconds(&self) -> f64 {
81        match self {
82            Time::Second(x) => *x,
83            Time::Year(x) => x * YEAR,
84            Time::Kiloyear(x) => x * YEAR * 1e3,
85            Time::Megayear(x) => x * YEAR * 1e6,
86            Time::Gigayear(x) => x * YEAR * 1e9,
87        }
88    }
89
90    /// Convert to gigayears
91    pub fn to_gyr(&self) -> f64 {
92        self.to_seconds() / GYR
93    }
94}

96/// Energy units
97#[derive(Debug, Clone, Copy, PartialEq)]
98pub enum Energy {
99    Joule(f64),
100    ElectronVolt(f64),
101    KiloElectronVolt(f64),
102    MegaElectronVolt(f64),
103    GeV(f64),
104}
105
106impl Energy {
107    /// Convert to joules
108    pub fn to_joules(&self) -> f64 {
109        match self {
110            Energy::Joule(x) => *x,
111            Energy::ElectronVolt(x) => x * EV_TO_J,
112            Energy::KiloElectronVolt(x) => x * EV_TO_J * 1e3,
113            Energy::MegaElectronVolt(x) => x * EV_TO_J * 1e6,
114            Energy::GeV(x) => x * EV_TO_J * 1e9,
115        }
116    }
117
118    /// Convert to eV
119    pub fn to_ev(&self) -> f64 {
120        self.to_joules() * J_TO_EV
121    }
122}

124/// Temperature conversion utilities
125pub mod temperature {
126    use super::*;
127
128    /// Convert temperature to energy (E = kT)
129    pub fn temp_to_energy_ev(temp_kelvin: f64) -> f64 {
130        (K_B * temp_kelvin) * J_TO_EV
131    }
132
133    /// Convert energy to temperature (T = E/k)
134    pub fn energy_to_temp(energy_ev: f64) -> f64 {
135        (energy_ev * EV_TO_J) / K_B
136    }
137}

139#[cfg(test)]
140mod tests {
141    use super::*;
142    use approx::assert_relative_eq;
143
144    #[test]
145    fn test_length_conversions() {
146        let d = Length::Megaparsec(1.0);
147        assert_relative_eq!(d.to_mpc(), 1.0, epsilon = 1e-10);
148        assert_relative_eq!(d.to_pc(), 1e6, epsilon = 1e-4);
149    }
150
151    #[test]
152    fn test_time_conversions() {
153        let t = Time::Gigayear(1.0);
154        assert_relative_eq!(t.to_gyr(), 1.0, epsilon = 1e-10);
155    }
156
157    #[test]
158    fn test_temperature_energy() {
159        let temp = 1e4; // 10,000 K
160        let energy = temperature::temp_to_energy_ev(temp);
161        let back = temperature::energy_to_temp(energy);
162        assert_relative_eq!(temp, back, epsilon = 1e-6);
163    }
